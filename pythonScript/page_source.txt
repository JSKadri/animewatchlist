	e.preventDefault();
	if (!confirm('This alias has never been the primary name for this tag')) {
		return;
	}
	var alt = $(e.target).attr('data-alt');
	var li = $(e.target).parent().parent();
	$.get('/delalt?alt=' + encodeURIComponent(alt) + '&ajax=1', function(e) {
		$('#alias-form').insertAfter('#aliases');
		li.slideUp(500).remove();
	});
});

		e.preventDefault();
	}
});
